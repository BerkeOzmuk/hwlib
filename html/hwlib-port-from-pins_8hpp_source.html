29 class port_in_out_from_pins : public port_in_out {
30 private:
31 
32  uint_fast8_t _number_of_pins;
33 
34  // 8 must match the number of parameters of the constructor
35  pin_in_out * pins[ 8 ];
36 
37 public:
38 
44  port_in_out_from_pins(
45  pin_in_out & p0 = pin_in_out_dummy,
46  pin_in_out & p1 = pin_in_out_dummy,
47  pin_in_out & p2 = pin_in_out_dummy,
48  pin_in_out & p3 = pin_in_out_dummy,
49  pin_in_out & p4 = pin_in_out_dummy,
50  pin_in_out & p5 = pin_in_out_dummy,
51  pin_in_out & p6 = pin_in_out_dummy,
52  pin_in_out & p7 = pin_in_out_dummy
53  ):
54  pins{ &p0, &p1, &p2, &p3, &p4, &p5, &p6, &p7 }
55  {
56  for( _number_of_pins = 0; _number_of_pins < 8; ++_number_of_pins ){
57  if( pins[ _number_of_pins ] == & pin_in_out_dummy ){
58  break;
59  }
60  }
61  }
62 
63  uint_fast8_t number_of_pins() override {
64  return _number_of_pins;
65  }
66 
67  void direction_set_input() override {
68  for( uint_fast8_t i = 0; i < _number_of_pins; i++ ){
69  pins[ i ]->direction_set_input();
70  }
71  }
72 
73  void direction_set_output() override {
74  for( uint_fast8_t i = 0; i < _number_of_pins; i++ ){
75  pins[ i ]->direction_set_output();
76  }
77  }
78 
79  uint_fast8_t read() override {
80  uint_fast8_t result = 0;
81  for( int_fast8_t i = _number_of_pins -1; i >=0; --i ){
82  result = result << 1;
83  if( pins[ i ]->read() ){
84  result |= 0x01;
85  }
86  }
87  return result;
88  }
89 
90  void refresh() override {
91  for( uint_fast8_t i = 0; i < _number_of_pins; i++ ){
92  pins[ i ]->refresh();
93  }
94  }
95 
96  void write( uint_fast8_t x ) override {
97  for( uint_fast8_t i = 0; i < _number_of_pins; i++ ){
98  pins[ i ]->write( ( x & 0x01 ) != 0 );
99  x = x >> 1;
100  }
101  }
102 
103  void flush() override {
104  for( uint_fast8_t i = 0; i < _number_of_pins; i++ ){
105  pins[ i ]->flush();
106  }
107  }
108
