685  void HWLIB_WEAK uart_putc_bit_banged_pin( char c, pin_out & pin ){
686  #ifdef BMPTK_TARGET
687  const auto bit_cel = ( ( 1000L * 1000L ) / BMPTK_BAUDRATE );
688 
689  pin.write( 1 );
690  // use busy waiting, otherwise logging from within the RTOS
691  // will cause serious problems
692  wait_us_busy( bit_cel );
693 
694  // start bit
695  pin.write( 0 );
696  wait_us_busy( bit_cel );
697 
698  // 8 data bits
699  for( uint_fast8_t i = 0; i < 8; ++i ){
700  pin.write( ( c & 0x01 ) != 0x00 );
701  c = c >> 1;
702  wait_us_busy( bit_cel );
703  }
704 
705  // 2 stop bits
706  pin.write( 1 );
707  wait_us_busy( 2 * bit_cel );
708  #else
709  #endif
710 
711  }

713  char HWLIB_WEAK uart_getc_bit_banged_pin( pin_in & pin ){
714  char c = 0;
715 
716  #ifdef BMPTK_TARGET
717  const auto bit_cel = ( ( 1000L * 1000L ) / BMPTK_BAUDRATE );
718 
719  // wait for start of startbit
720  while( pin.read() ){ wait_ns( 1 ); }
721 
722  // wait until halfway the first data bit
723  auto t = now_us();
724  t += bit_cel + ( bit_cel / 2 );
725  while( now_us() < t ){};
726 
727  // 8 data bits
728  for( uint_fast8_t i = 0; i < 8; ++i ){
729  c = c >> 1;
730  if( pin.read() ){
731  c = c | 0x80;
732  }
733 
734  t+= bit_cel;
735  while( now_us() < t ){};
736  }
737 
738  #else
739  #endif
740  return c;
741  }
