196 void HWLIB_NORETURN graphics_random_lines(
197  window & w,
198  uint_fast16_t n_lines
199 ){
200  for(;;){
201  w.clear();
202  for( uint_fast16_t x = 0; x < w.size.x; x++ ){
203  w.write( hwlib::location{
204  x,
205  0 } );
206  w.write( hwlib::location{
207  x,
208  static_cast< uint_fast16_t >( w.size.y - 1 ) } );
209  }
210  for( uint_fast16_t y = 0; y < w.size.y; y++ ){
211  w.write( hwlib::location{
212  0,
213  y } );
214  w.write( hwlib::location{
215  static_cast< uint_fast16_t >( w.size.x - 1 ),
216  y } );
217  }
218  w.flush();
219  for( uint_fast16_t n = 0; n < n_lines; n++ ){
220  const uint_fast16_t x = random_in_range( 0, w.size.y );
221  const uint_fast16_t x1 = random_in_range( 0, w.size.x );
222  const uint_fast16_t y = random_in_range( 0, w.size.y );
223  const uint_fast16_t y1 = random_in_range( 0, w.size.y );
224  auto line = hwlib::line(
225  hwlib::location{ x, y },
226  hwlib::location{ x1, y1 },
227  w.foreground
228  );
229  line.draw( w );
230  w.flush();
231  hwlib::wait_ms( 500 );
232  }
233  }
234 }
