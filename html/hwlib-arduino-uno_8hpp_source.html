428 void HWLIB_WEAK wait_us_asm( int n ){
429  // first int parameter is passd in r24/r25
430  __asm volatile( // clocks
431  "1: cp r1, r24 \t\n" // 1
432  " cpc r1, r25 \t\n" // 1
433  " brge 3f \t\n" // 1
434  " rcall 3f \t\n" // 7
435  " rjmp 2f \t\n" // 2
436  "2: sbiw r24, 0x01 \t\n" // 2
437  " rjmp 1b \t\n" // 2
438  "3: \t\n" // 16 total
439  : : "r" ( n ) // uses (reads) n
440  );
441 
442 }

444 void HWLIB_WEAK wait_us( int_fast32_t n ){
445  while( n > 0 ){
446  if( n < 10'000 ){
447  wait_us_asm( n );
448  return;
449  }
450  wait_us_asm( 1'000 );
451  n -= 1'000;
452  }
453 }
