52  uint_fast8_t get( uint_fast8_t channel ){
53  // select the correct channel
54  uint8_t control = ( configuration & ( ~ 0x03 )) + channel;
55  bus.write( base + address, & control, 1 );
56 
57  // read results, note that the first byte is the
58  // *previous* ADC result, the second byte is what we want
59  uint8_t results[ 2 ];
60  bus.read( base + address, results, 2 );
61  return results[ 1 ];
62  }
