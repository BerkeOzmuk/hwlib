116  // configure a pin as ADC
117  uint32_t __attribute__((weak)) configure_as_adc( uint32_t port, uint32_t pin ){
118 
119  // enable IOCON & A/D clock
120  LPC_SYSCON->SYSAHBCLKCTRL |= ( 0x01 << 16 ) | ( 0x01 << 13 );
121 
122  // enable A/D power
123  LPC_SYSCON->PDRUNCFG &= ~( 1 << 4 );
124 
125  switch( ( port << 8 ) | pin ){
126  case 0x000B: pin_ad_conf( &LPC_IOCON->R_PIO0_11, 0x02 ); return 0;
127 
128  case 0x0100: pin_ad_conf( &LPC_IOCON->R_PIO1_0, 0x02 ); return 1;
129  case 0x0101: pin_ad_conf( &LPC_IOCON->R_PIO1_1, 0x02 ); return 2;
130  case 0x0102: pin_ad_conf( &LPC_IOCON->R_PIO1_2, 0x02 ); return 3;
131  case 0x0103: pin_ad_conf( &LPC_IOCON->SWDIO_PIO1_3, 0x02 ); return 4;
132  case 0x0104: pin_ad_conf( &LPC_IOCON->PIO1_4, 0x01 ); return 5;
133 
134  default: HWLIB_PANIC_WITH_LOCATION;
135  }

278 class pin_adc : public hwlib::adc {
279 private:
280  uint32_t channel;
281 
282 public:
283 
285  pin_adc( uint32_t port, uint32_t pin )
286  : adc{ 10 }, channel{ configure_as_adc( port, pin ) }
287  {}
288 
289  adc_value_type get() override {
290 
291  // configure the A/D for the pin
292  LPC_ADC->CR = ( 0x01 << channel ) | ( 12 << 8 );
293 
294  // start the conversion
295  LPC_ADC->CR |= ( 0x01 << 24 );
296 
297  // wait for the conversion to complete
298  while( ( LPC_ADC->GDR & ( 1 << 31 )) == 0 );
299 
300  hwlib::wait_ms( 10 );
301 
302  // return the A/D result
303  return 3 | (( LPC_ADC->GDR >> 6 ) & 0x3FF);
304  }

318 void HWLIB_WEAK wait_us( int_fast32_t n ){
319  // for the default clock frequency of 12 MHz:
320  // the loop uses 12 cycles
321  __asm volatile(
322  " mov r0, %[reg] \t\n"
323  " b 1f \t\n"
324  " .align 4 \t\n"
325  "1: sub r0, #1 \t\n"
326  " b 2f \t\n"
327  "2: b 3f \t\n"
328  "3: nop \t\n"
329  " nop \t\n"
330  " bgt 1b \t\n"
331  :: [reg] "r" (n) : "r0"
332  );
333 }
